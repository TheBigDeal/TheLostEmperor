scripted_diplomatic_actions = {
	offer_alliance = {
		icon = 16
		show_acceptance_on_action_button = yes
		requires_acceptance = yes
		allowed = {
			always = yes
		}
		visible = {
			is_subject = no
		}
		on_sent_effect = {
			ROOT = { set_country_flag = { flag = diplomatic_action_cooldown value = 1 days = 7 } }
		}
		complete_effect = { #Positive Relations + Alliance
			#
		}
		reject_effect = { # Negative Relations
		
		}
		ai_desire = {
			modifier = {
				add = 20
				Condition to add above
				ROOT = {
					has_opinion = {
						target = FROM
						value > 100
					}
				}
			}
			modifier = {
				add = -200
				ROOT = { has_country_flag = diplomatic_action_cooldown }
			}
		}
		ai_acceptance = {
			base_desire = {
				modifier = {
					factor = -40
				}
			}
			stronger_nation = {
				modifier = {
					factor = 50
					FROM = {
						strength_ratio = {
							tag = ROOT
							ratio < 0.35
						}
					}
				}
			}
			opinion_desire = {
				modifier = {
					factor = FROM.opinion@ROOT
					FROM = {
						has_opinion = {
							target = ROOT
							value > 0
						}
					}
				}
				modifier = {
					factor = FROM.opinion@ROOT
					FROM = {
						has_opinion = {
							target = ROOT
							value < 0
						}
					}
				}
			}
			too_weak = {
				modifier = {
					factor = -20
					FROM = {
						strength_ratio = {
							tag = ROOT
							ratio > 0.65
						}
					}
				}
				modifier = {
					factor = 0
					FROM = {
						strength_ratio = {
							tag = ROOT
							ratio < 0.65
						}
					}
				}
			}
			subject_nation = {
				modifier = {
					factor = -200
					FROM = {
						is_subject = yes
					}
				}
				modifier = {
					factor = 0
					FROM = {
						is_subject = no
					}
				}
			}
		}
	}
	enforce_claims = {
		show_acceptance_on_action_button = no
		requires_acceptance = no
		allowed = {
			always = yes
		}
		visible = {
			is_subject = no
		}
		selectable = { #PREV = Target Country
			PREV = { 
				any_owned_state  = {
					limit = {
						is_owned_by = PREV
					}
					is_claimed_by = ROOT
				}
			}
		}
		cost = 150
		complete_effect = {
			ROOT = { create_wargoal = { type = annex_everything target = PREV } }
		}
	}
	impose_trade_embargo = {

		allowed = {
			always = yes
		}
		visible = {
			ROOT = {
				is_subject = no
				NOT = {
					has_relation_modifier = {
						target = PREV
						modifier = embargo_static_modifier
					}
				has_idea = closed_economy
				is_subject_of = PREV
				}
			}
			PREV = {
				NOT = { is_subject_of = ROOT }
			}
		}

		selectable = {
			ROOT = {
				strength_ratio = {
					tag = PREV
					ratio > 0.50
				}
				
			}
			IF = {
				limit = {
					PREV = {
						OR = {
						has_war_together_with = ROOT 
						is_in_faction_with = ROOT 
						has_non_aggression_pact_with = ROOT 
						has_war_with = ROOT
						
						}
					}
				}
			}
			else = {
				custom_trigger_tooltip = {
					tooltip = EMBARGO_ALLOWED
					always = yes
				}
			}
		}

		cost = 150
		command_power = 0

		show_acceptance_on_action_button = no

		requires_acceptance = no

		accept_title = IMPOSE_EMBARGO_TARGET_TITLE
		accept_description = IMPOSE_EMBARGO_TARGET_DESC

		can_be_sent = {
			always = yes
		}

		icon = 1

		complete_effect = {
			impose_embargo = yes
		}

		ai_desire = {
			base = 1
			modifier = { #game rule
				has_game_rule = {
					rule = allow_ai_embargo
					option = DONT_ALLOW_AI_EMBARGO
				}
			add = -5000
			}
			modifier = { #If opinion is lower than -50 they will impose embargo if no puppet 
				ROOT = {
					AND = {
						is_puppet = no
						has_opinion = {
							target = PREV
							value < -50
						}
					}
				}
			add = 50
			}

			modifier = { #If opinion is higher than -50 they won't impose embargo if no puppet
				ROOT = {
					AND = {
						is_puppet = no
						has_opinion = {
							target = PREV
							value > -50
						}
					}
				}
			add = -50
			}
			modifier = { #Counter Embargo
				PREV = {
					has_relation_modifier = {
						target = ROOT
						modifier = embargo_static_modifier
					}
				}
			add = 100
			}
			modifier = { #if opinion is higher than 50 they'll never impose embargo
				ROOT = {
					has_opinion = {
						target = PREV
						value > 49
					}
				}
			add = -100
			}
			modifier = { #if ally is at war with target they'll impose embargo regardless of autonomy state 
				ROOT = {
					any_allied_country = {
						has_war_with = PREV.PREV
					}
				}
				add = 300
			}

			modifier = { #if puppet they'll follow master #fixed DLC use now without any DLC
						AND = {
							ROOT = {
								is_puppet = yes
							}
							any_other_country = {
								has_subject = ROOT
								has_relation_modifier = {
									target = PREV
									modifier = embargo_static_modifier
								}
							}
						}
			add = 300
			}
		}
	}

	cancel_trade_embargo = {

		allowed = {
			always = yes
		}

		visible = {
			ROOT = {
				has_relation_modifier = {
				target = PREV
				modifier = embargo_static_modifier
				}
			}
		}

		selectable = {
			IF = {
				limit = {
					ROOT = {
						has_relation_modifier = { target = PREV modifier = embargo_static_modifier }
					}
					PREV = {
						AND = {
							OR = {
							is_subject_of = ROOT
							NOT = { has_war_with = ROOT }
							}
						}
					}
				}
				custom_trigger_tooltip = {
					tooltip = CANCEL_EMBARGO_ALLOWED
					always = yes
				}
			}
			else = {
				custom_trigger_tooltip = {
					tooltip = CANCEL_EMBARGO_DENIED
					always = no
				}
			}
		}

		cost = 10
		command_power = 0

		requires_acceptance = no

		show_acceptance_on_action_button = no

		accept_title = CANCEL_EMBARGO_TARGET_TITLE
		accept_description = CANCEL_EMBARGO_TARGET_DESC

		can_be_sent = {
			always = yes
		}

		icon = 1

		complete_effect = {
			cancel_embargo = yes
		}

		ai_desire = {
			base = 1

			modifier = { #game rule
				has_game_rule = {
					rule = allow_ai_embargo
					option = DONT_ALLOW_AI_EMBARGO
				}
			add = -5000
			}
			modifier = { #if opinion is higher -50 they'll cancel emb if not puppet
				ROOT = {
					AND = {
						is_puppet = no
						has_opinion = {
							target = PREV
							value > -50
						}
					}
				}
			add = 50
			}
			modifier = { #if opinion is lower -50 they won't cancel if not puppet
				ROOT = {
					AND = {
						is_puppet = no
						has_opinion = {
							target = PREV
							value < -50
						}
					}
				}
			add = -50
			}
			modifier = { #if ally has war with target they won't cancel
				ROOT = {
					any_allied_country = {
						has_war_with = PREV.PREV
					}
				}
			add = -300
			}

			modifier = { #see above #won't cancel if master has embargo
						AND = {
							ROOT = {
								is_puppet = yes
							}
							any_other_country = {
								has_subject = ROOT
								has_relation_modifier = {
									target = PREV
									modifier = embargo_static_modifier
								}
							}
						}
			add = -300
			}

			modifier = { #this is used to lift embargo when master has lifted
				AND = {
					ROOT = {
						AND = {
							is_puppet = yes
							has_relation_modifier = {
								target = PREV
								modifier = embargo_static_modifier
							}
						}
					}
					any_other_country = {
						has_subject = ROOT
						NOT = {
							has_relation_modifier = {
								target = PREV
								modifier = embargo_static_modifier
							}
						}
					}
				}
			add = 300
			}

			modifier = { #Counter Embargo
				PREV = {
					has_relation_modifier = {
						target = ROOT
						modifier = embargo_static_modifier
					}
				}
			add = -300
			}
		}
	}
}
