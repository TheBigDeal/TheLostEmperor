on_actions = {
	on_startup = {
		effect = {
			set_variable = { var = GLOBAL_COMMUNIST_INFLUENCE var_name value = 0 }
			set_variable = { var = GLOBAL_FASCIST_INFLUENCE var_name value = 0 }
			set_variable = { var = GLOBAL_THEOCRATIC_INFLUENCE var_name value = 0 }
			set_variable = { var = GLOBAL_DEMOCRATIC_INFLUENCE var_name value = 0 }
			#64 = { set_state_flag = GERMAN_CONFEDERATION }
			# num_battalions max_manpower_k
			every_country = {
				set_variable = { power_rating }
				set_variable = { temp_bat = num_battalions }
				set_variable = { temp_pop = max_manpower_k }
				divide_variable = {
					var = temp_pop
					value = 1000
				}
				divide_variable = {
					var = temp_bat
					value = 20
				}
				add_to_variable = {
					var = power_rating
					value = num_of_factories
				}
				add_to_variable = {
					var = power_rating
					value = temp_bat
				}
				add_to_variable = {
					var = power_rating
					value = temp_pop
				}
			}
		}
	}
	on_daily = {
		effect = {
			every_country = {
				clear_variable = power_rating
				clear_variable = temp_bat
				clear_variable = temp_pop
				set_variable = { power_rating }
				set_variable = { temp_bat = num_battalions }
				set_variable = { temp_pop = max_manpower_k }
				divide_variable = {
					var = temp_pop
					value = 1000
				}
				divide_variable = {
					var = temp_bat
					value = 20
				}
				add_to_variable = {
					var = power_rating
					value = num_of_factories
				}
				add_to_variable = {
					var = power_rating
					value = temp_bat
				}
				add_to_variable = {
					var = power_rating
					value = temp_pop
				}
			}
		}
	}
}
