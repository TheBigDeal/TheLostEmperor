on_actions = {
	on_startup = {
		effect = {
			country_event = { id = great_power.2 }
			set_country_flag = show_economic_tab
			sort_power_ratings = yes
			every_country = { refresh_economy = yes }
			set_variable = { current_money = 0 }
			reload_ideas = yes
		} 
	}
    
    on_daily = {
		effect = {
			for_each_loop = {
				if = {
					limit = { check_variable = { v = 7 } }
					every_country = { refresh_economy = yes }
					add_to_variable = { current_money = current_money_rate }	
					round_variable = current_money				
					clamp_variable = { var = current_money min = -999999 max = 999999 }
				}
				set_temp_variable = { break = 7 }
			}
        }
    }
}
