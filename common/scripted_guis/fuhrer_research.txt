scripted_gui = {

	current_money_research = {
		context_type = player_context
		
		window_name = "current_money_research_tab"
		parent_window_token = technology_tab
        
		visible = {
            has_current_money = yes
		}
		
		effects = {
            current_money_boost_research_click = {
                add_ideas = fuhrer_boost_research 
                subtract_from_variable = { current_money = 15 }
			}
            current_money_boost_research_cancel_click = {
				swap_ideas = {
					remove_idea = fuhrer_boost_research
				}
			}
		}
        
        
		triggers = {
            current_money_boost_research_click_enabled = {
                NOT = { has_idea =  fuhrer_boost_research  }
                check_variable = { current_money > 14.999 }
			}
            current_money_boost_research_visible = {
                NOT = { has_idea =  fuhrer_boost_research  }
			}
            current_money_boost_research_cancel_visible = {
                has_idea =  fuhrer_boost_research
			}
		}
	}
}
