scripted_gui = {
	current_money = {
		context_type = player_context
		window_name = "current_money_container"
		visible = { always = yes }
		effects = {
			current_money_bg_click = {
				if = {
					limit = {
						check_variable = { open_economic_window = 1 }
					}
					set_variable = { open_economic_window = 0 }
				}
				else = {
					set_variable = { open_economic_window = 1 }
				}
			}
		}
	}
}
