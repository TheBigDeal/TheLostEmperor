country_event = {
	id = political.9
	title = political.9.t
	desc = political.9.d
	picture = GFX_report_event_worker_protests
	trigger = {
		OR = {
			has_idea_with_trait = labryscist_demagogue
			has_idea_with_trait = rexist_demagogue
		}
		AND = {
			NOT = { has_government = labryscism }
			NOT = { has_government = rexism }
		}
		OR = {
			labryscism > 0.25
			rexism > 0.25
		}
		any_neighbor_country = {
			has_government = collectivism
		}
	}
	mean_time_to_happen = {
		days = 365
		modifier = {
			factor = 0.5
			OR = {
				has_country_flag = labryscism_popular_movement
				has_country_flag = rexism_popular_movement
			}
		}
	}
	option = {
		name = political.9.a
		if = {
			limit = {
				labryscism > 0.25
			}
			add_popularity = {
				ideology = labryscism
				popularity = 0.1
			}
		}
		if = {
			limit = {
				rexism > 0.25
			}
			add_popularity = { ideology = rexism popularity = 0.1 }
		}
	}
}

		OR = {
			has_idea_with_trait = liberalism_reformer
			has_idea_with_trait = conservatism_reformer
			has_idea_with_trait = socialism_reformer
		}
		OR = {
			NOT = { has_government = liberalism }
			NOT = { has_government = conservatism }
			NOT = { has_government = socialism }
		}
		OR = {
			has_country_flag = liberal_opposition_formed
			has_country_flag = socialist_opposition_formed
			has_country_flag = convervative_opposition_formed
		}
		AND = {
			NOT = { has_idea = liberalism_opposition_voicing_protests }
			NOT = { has_idea = socialism_opposition_voicing_protests }
			NOT = { has_idea = conservatism_opposition_voicing_protests }
		}
	}
	mean_time_to_happen = {
		days = 150
	}
	option = {
		name = political.15.a
	}
	option = {
		name = political.15.b
		add_political_power = -20
		if = {
			limit = {
				has_idea_with_trait = liberalism_reformer
			}
			add_ideas = liberalism_opposition_voicing_protests
		}
		if = {
			limit = {
				has_idea_with_trait = conservatism_reformer
			}
			add_ideas = conservatism_opposition_voicing_protests
		}
		if = {
			limit = {

			}

		}
	}
}

country_event = { #Labryscist Revolution
	id = political.25
	title = political.25.t
	desc = political.25.d
	picture = GFX_report_event_fascist_militia
	trigger = {
		labryscism > 0.7
		#has_idea = labryscist_revolutionaries
		NOT = { has_government = labryscism }
		has_civil_war = no
		#check_has_focus_tree_to_switch_to_labryscism = yes
	}
	mean_time_to_happen = { days = 2 }
	immediate = {
		hidden_effect = { remove_ideas = labryscist_revolutionaries }
	}
	option = { #Stay with current government
		name = political.25.a
		ai_chance = {
			factor = 0
		}
		start_civil_war = { ideology = labryscism size = 0.5 }
	}
	option = { #Side with the Revolutionaries
		name = political.25.b
		#Democratic#
		###################################
		if = {
			limit = { has_government = liberalism }
			start_civil_war = { ruling_party = labryscism ideology = liberalism size = 0.5 }
		}
		if = {
			limit = { has_government = conservatism }
			start_civil_war = { ruling_party = labryscism ideology = conservatism size = 0.5 }
		}
		if = {
			limit = { has_government = socialism }
			start_civil_war = { ruling_party = labryscism ideology = socialism size = 0.5 }
		}
		#Communist#
		###################################
		if = {
			limit = { has_government = collectivism }
			start_civil_war = { ruling_party = labryscism ideology = collectivism size = 0.5 }
		}
		if = {
			limit = { has_government = mutualism }
			start_civil_war = { ruling_party = labryscism ideology = mutualism size = 0.5 }
		}
		if = {
			limit = { has_government = proletarianism }
			start_civil_war = { ruling_party = labryscism ideology = proletarianism size = 0.5 }
		}
		#Monarchic#
		###################################
		if = { 
			limit = { has_government = absolutism }
			start_civil_war = { ruling_party = labryscism ideology = absolutism size = 0.5 }
		}
		if = { 
			limit = { has_government = theocracy }
			start_civil_war = { ruling_party = labryscism ideology = theocracy size = 0.5 }
		}
		if = { 
			limit = { has_government = royalism }
			start_civil_war = { ruling_party = labryscism ideology = royalism size = 0.5 }
		}
		#Fascism#
		###################################
		if = { 
			limit = { has_government = labryscism }
			start_civil_war = { ruling_party = labryscism ideology = labryscism size = 0.5 }
		}
		if = { 
			limit = { has_government = rexism }
			start_civil_war = { ruling_party = labryscism ideology = rexism size = 0.5 }
		}
		if = { 
			limit = { has_government = despotism }
			start_civil_war = { ruling_party = labryscism ideology = despotism size = 0.5 }
		}
	}
}
