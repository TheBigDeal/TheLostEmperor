add_namespace = le_generic

country_event = { #Minor Exceeding Force Limit
	id = le_generic.1
	title = le_generic.1.t
	desc = le_generic.1.d
	picture = GFX_report_event_generic_read_write
	trigger = {
		NOT = { has_country_flag = minor_exceeding_force_limit }
		check_variable = { num_divisions > force_limit_6 }
	}
	option = {
		name = le_generic.1.a
		set_country_flag = minor_exceeding_force_limit
		add_ideas = {
			minor_exceeded_force_limit
		}
	}
}
country_event = { #Exceeding Force Limit
	id = le_generic.2
	title = le_generic.2.t
	desc = le_generic.2.d
	picture = GFX_report_event_generic_read_write
	trigger = {
		NOT = { has_country_flag = exceeding_force_limit }
		has_country_flag = minor_exceeding_force_limit
		check_variable = { num_divisions > force_limit_12 }
	}
	option = {
		name = le_generic.2.a
		set_country_flag = exceeding_force_limit
		if = {
			limit = { has_idea = minor_exceeded_force_limit}
			swap_ideas = { remove_idea = minor_exceeded_force_limit add_idea = exceeded_force_limit}
		}
		else = {
			add_ideas = { exceeded_force_limit }
		}
	}
}
country_event = { #Majorly Exceeding Force Limit
	id = le_generic.3
	title = le_generic.3.t
	desc = le_generic.3.d
	picture = GFX_report_event_generic_read_write
	trigger = {
		NOT = { has_country_flag = major_exceeding_force_limit }
		has_country_flag = minor_exceeding_force_limit
		has_country_flag = exceeding_force_limit
		check_variable = { num_divisions > force_limit_18 }
	}
	option = {
		name = le_generic.3.a
		set_country_flag = major_exceeding_force_limit
		if = {
			limit = { has_idea = minor_exceeded_force_limit}
			swap_ideas = { remove_idea = minor_exceeded_force_limit add_idea = majorly_exceeded_force_limit}
		}
		else_if = {
			limit = { has_idea = exceeded_force_limit}
			swap_ideas = { remove_idea = exceeded_force_limit add_idea = majorly_exceeded_force_limit}
		}
		else = {
			add_ideas = { majorly_exceeded_force_limit }
		}
	}
}
country_event = { #Majorly Exceeding Force Limit
	id = le_generic.4
	title = le_generic.4.t
	desc = le_generic.4.d
	picture = GFX_report_event_generic_read_write
	trigger = {
		OR = {
			has_country_flag = minor_exceeding_force_limit
			has_country_flag = exceeding_force_limit
			has_country_flag = major_exceeding_force_limit
		}
		check_variable = { num_divisions < force_limit }
	}
	option = {
		name = le_generic.4.a
		clr_country_flag = minor_exceeding_force_limit
		clr_country_flag = exceeding_force_limit
		clr_country_flag = major_exceeding_force_limit
		if = {
			limit = { has_idea = majorly_exceeded_force_limit}
			remove_ideas = majorly_exceeded_force_limit
		}
		else_if = {
			limit = { has_idea = exceeded_force_limit}
			remove_ideas = exceeded_force_limit
		}
		else_if = {
			limit = { has_idea = minor_exceeded_force_limit}
			remove_ideas = minor_exceeded_force_limit
		}
	}
}
